import React from 'react';
import QRCode from 'qrcode.react';

const TransactionQRCode = ({ transactionDetails }) => {
  const qrData = JSON.stringify(transactionDetails); // Prepare data for QR code

  return (
    <div>
      <h2>Scan to Pay</h2>
      <QRCode value={qrData} size={256} />
      <p>Scan the QR code to complete your transaction.</p>
    </div>
  );
};

export default TransactionQRCode;
const transactionDetails = {
    toAccountId: 3,
    amount: 20,
    remark: "Payment for services"
  };
  
  const App = () => {
    return (
      <div>
        <h1>Transaction QR Code</h1>
        <TransactionQRCode transactionDetails={transactionDetails} />
      </div>
    );
  };
  
  export default App;
